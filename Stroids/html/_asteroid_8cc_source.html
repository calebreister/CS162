97 bool Asteroid::checkSide(Side s)
98 {
99  /*
100  * NOTE: this is not the simplest method, but the other requires extra functions,
101  * passing a location COMPONENT that is correct (*.y for top/bottom, *.x for left/right,
102  * passing the entire location vector would be pointless for individual functions).
103  * This method is also more self-documenting. In addition, no mathematical evaluation is
104  * required when checking a side as it was already passed in the parameter. In addition,
105  * the switch is a very streamlined method of writing this code, any other method would
106  * require significantly more unnecessary whitespace. Based off of my understanding,
107  * this is the fastest method.
108  */
109 
110  static Vect2d loc; //static prevents memory from having to be re-allocated repeatedly
111  loc = getLocation();
112  switch (s)
113  {
114  case left:
115  return (loc.x - getRadius() < ZERO.x) ? true : false;
116  break;
117  case right:
118  return (loc.x + getRadius() > WIN_SIZE.x) ? true : false;
119  break;
120  case top:
121  return (loc.y - getRadius() < ZERO.y) ? true : false;
122  break;
123  case bottom:
124  return (loc.y + getRadius() > WIN_SIZE.y) ? true : false;
125  break;
126  }
127  return false;
128 }
